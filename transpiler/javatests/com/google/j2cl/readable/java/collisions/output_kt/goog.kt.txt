  val T_1: Int = 5

}

open class T<T: kotlin.Number?> {
 internal constructor(): super()

 internal open fun m(): T? {
  return null
 }

 internal open fun n(): T? {
  val t: T? = collisions.T<T>().m()
  (t!! as java.lang.Number).byteValue()
  return t
 }
}
