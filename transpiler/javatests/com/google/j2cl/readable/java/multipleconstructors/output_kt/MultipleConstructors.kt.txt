package multipleconstructors

open class MultipleConstructors {
 internal var id: Int = 0

 internal var flag: Boolean = false

 constructor(id: Int) {
  this.id = id
  this.flag = id == 0
 }

 constructor(flag: Boolean) {
  this.id = - 1
  this.flag = flag
 }

 constructor(id: Int, flag: Boolean) {
  this.id = id
  this.flag = flag
 }
}
