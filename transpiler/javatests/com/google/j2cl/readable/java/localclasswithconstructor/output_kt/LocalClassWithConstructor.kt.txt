open class LocalClassWithConstructor {
 open fun test(p: Int) {
  val localVar: Int = 1
  open class `1LocalClass` {
   @kotlin.jvm.JvmField var field: Int = 0

   constructor(a: Int, b: Int) {
    this.field = localVar + a + b
   }

   constructor(a: Int): this(a, p) {
    this.field = localVar
   }
  }
  `1LocalClass`(1)
 }
}
