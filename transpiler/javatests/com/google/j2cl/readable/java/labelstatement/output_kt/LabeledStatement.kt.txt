package labelstatement

import javaemul.lang.*
import kotlin.jvm.*

open class LabeledStatement {
 open fun loopStatements() {
  LABEL@ while (true) {
   LABEL_CONTINUE@ do {
    break@LABEL
   } while (false)
  }
 }

 open fun simpleStatement() {
  LABEL@ do {
   this.foo()
  } while (false)
  LOOP@ do {
   LABEL_1@ do {
    continue@LOOP
   } while (false)
  } while (false)
  LABEL_2@ do {
   return
  } while (false)
 }

 open fun block() {
  LABEL@ do {
   break@LABEL
  } while (false)
 }

 open fun ifStatement() {
  LABEL@ do {
   if (true) {
    break@LABEL
   }
  } while (false)
 }

 open fun switchStatement() {
  LABEL@ do {
   CASE@ do {
    when (0) {
     0 -> {
      break@CASE
     }
    }
    break@LABEL
   } while (false)
   if (true) {
    break@LABEL
   }
  } while (false)
 }

 fun foo() {}
}
